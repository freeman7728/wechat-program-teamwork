int a[20];

int n,k,sum,ans;

bool check(int num) {
    if (num == 1) {
        return false;
    }
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return false;
        }
    }
    return true;
}

int dfs(int d,int start)
{
	if(d==k+1)
	{
	   int sum=0;
	   for(int i=1;i<=k;i++)
	   {
	   	 cout<<c[i]<<' ';（这步只是我验证排列是否正确)
	   	 sum+=c[i];
	   } 
	   cout<<'\n';
	   if(check(sum)) ans++;
	}
	for(int i=start;i<=n;i++)
	{
		c[d]=a[i];
	    dfs(d+1,i+1);
	}
	return ans;
}
